package user

class User {
    string name
    int age
    string cpf

    constructor(string name, int age, string cpf) {
        this.name = name
        this.age = age
        this.cpf = cpf
    }

    string method cpf() {
        return this.cpf
    }

    generic<T> (string, int, T) method values(T generico) {
        return this.name, this.age, generico
    }
}

User user = new User("Samuel", 15, "111-111-111.11")
var cpf = user.cpf()
var name, age, num = generic<int> user.values(1000)

export User
