Recursive Fibonacci

Demonstration of functions and control structures with recursion.

math.al
func fib(n: int) -> int {
  if (n <= 1) return n
  return fib(n - 1) + fib(n - 2)
}

func main() {
  io.println(fib(10)) // Output: 55
}

Minimal HTTP Server

Build a high-performance web server using the `alpha.net` library.

server.al
import alpha.net

func handler(req: Request, res: Response) {
  res.status(200).send("Alpha Server is running!")
}

func main() {
  let server = net.NewServer(":8080")
  server.route("/", handler)
  server.listen()
}
